def ask():
    while True:
        cords = input("         Ваш ход: ").split()

        if len(cords) != 2:
            print(" Введите 2 координаты. ")
            continue

        x, y = cords

        if not (x.isdigit()) or not (y.isdigit()):
            print(" Введите числа. ")
            continue

        x, y = int(x), int(y)

        if 0 > x or x > 2 or 0 > y or y > 2:
            print(" Вводите координаты соответствующие игровому полю! ")
            continue

        if field[x][y] != " ":
            print(" Клетка занята! ")
            continue

        return x, y



def check_win():

            print("Выиграл 0")
            return True
    for i in range(3):
        symbols = []
        for j in range(3):
            symbols.append(field[j][i])
        if symbols == ["X", "X", "X"]:
            print("Выиграл X")
            return True
        if symbols == ["0", "0", "0"]:
            print("Выиграл 0")
            return True
    symbols = []
    for i in range(3):

    if symbols == ["X", "X", "X"]:
        print("Выиграл X")
        return True
    if symbols == ["0", "0", "0"]:
        print("Выиграл 0")
        return True

    symbols = []

        symbols.append(field[i][2-i])

        print("Выиграл X")
        return True

        print("Выиграл 0")
        return True
